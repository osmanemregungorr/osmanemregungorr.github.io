2. Ortam İzleme — ARDUİNO UNO

Blok Şeması

Bacak Bağlantı Şeması

Projenin Amacı

Bu proje, çoklu sensörden gelen veriyi eşzamanlı olarak işleyip görselleştiren bir izleme sistemidir. Buradaki asıl teknik zorluk, birden fazla sensörü birbirini bloklamadan okuyabilmektir.

Nasıl Çalışır?

  • 1. Çoklu Sensör Okuma: DHT22 (sıcaklık/nem), LDR (ışık) ve MQ-135 (hava kalitesi) sensörlerinden veri alınır. Her sensörün farklı örnekleme hızı vardır.
  • 2. Veri İşleme (Non-blocking): `delay()` yerine `millis()` kullanılarak sistemin sürekli aktif kalması ve aynı anda birden fazla işi yönetmesi sağlanır.
  • 3. Görselleştirme: Veriler I2C protokolü ile 0.96" OLED ekranda anlık olarak gösterilir.
  • 4. Eşik Bazlı Uyarı: Değerler normal aralık dışına çıkarsa RGB LED renk değiştirir ve buzzer sesli uyarı verir.

Öğretici Değer

  • I2C Haberleşme: Aynı SDA/SCL hattı üzerinden farklı adresli cihaz yönetimi.
  • Multitasking Mantığı: Gömülü sistemlerde kesintisiz çalışma için zamanlayıcı (timer) kullanımı.

#include

#include //D2 PİNİNİ KÜTÜPHANE İLE HEM OUTPUT HEM DE İNPUT YAPAR.

#define mq_135 A1
#define dht_22 2
#define dht_tipi DHT22
#define ldr A0
#define kirmizi 9
#define yesil 10
#define mavi 11
#define buzzer 6

#define ekran_genislik 128
#define ekran_yukseklik 64
#define ekran_reset -1

DHT dht(dht_22, dht_tipi);
Adafruit_SSD1306 display(ekran_genislik, ekran_yukseklik, &Wire, ekran_reset);

unsigned long son_ldr_okumasi=0;
const unsigned long ldr_okuma_araligi=200;
int anlik_isik=0;

unsigned long son_dht_okumasi=0;
const unsigned long dht_okuma_araligi=2500;
float sicaklik=0.0;
float nem=0.0;

float min_nem=30.0;
float max_nem=80.0;
float min_sicaklik=17.0;
float max_sicaklik=33.0;
bool alarm=false;
bool sensor_hata=false;

unsigned long alarm_ses=0;
const unsigned long alarm_ses_araligi=1000;

void renk_sec(int k,int y,int m){
analogWrite(kirmizi,k);
analogWrite(yesil,y);
analogWrite(mavi,m);
}

void setup() {
  dht.begin();
  Serial.begin(9600);

  display.clearDisplay(); // Ekran hafızasını temizleme
  display.setTextColor(SSD1306_WHITE); // Yazı rengini beyaz yapma
  display.setTextSize(1); // Varsayılan yazı boyutunu 1 yapma

  pinMode(kirmizi,OUTPUT);
  pinMode(mavi,OUTPUT);
  pinMode(yesil,OUTPUT);
  pinMode(buzzer,OUTPUT);
}

void loop() {
  unsigned long simdiki_zaman=millis();
  if(simdiki_zaman-son_ldr_okumasi>=ldr_okuma_araligi){
    son_ldr_okumasi=simdiki_zaman;
    anlik_isik=map(analogRead(ldr),0,1023,0,100);
    Serial.println("Alık Işık:");
    Serial.print(anlik_isik);
  }
  if(simdiki_zaman-son_dht_okumasi>=dht_okuma_araligi){
    float s=dht.readTemperature();
    float n=dht.readHumidity();
    son_dht_okumasi=simdiki_zaman;
    if(s==s&&n==n){
      sensor_hata=false;
      sicaklik=s;
      nem=n;
      Serial.print("Sıcaklık: ");
      Serial.println(s);
      Serial.print("Nem: ");
      Serial.println(n);
    }
    else{
      sensor_hata=true;
    }
  }
  if(sensor_hata==true){
    renk_sec(0,0,255);
  }
  else{
    if((max_nem >nem || nemsicaklik || sicaklik=alarm_ses_araligi){
      alarm_ses=simdiki_zaman;
      tone(buzzer,1500,150);
    }
  }
  else{
    renk_sec(0,255,0);
  }
  }
}
