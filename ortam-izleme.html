2. Ortam İzleme — Arduino Uno

Projenin Amacı

Bu proje, çoklu sensörden gelen veriyi eşzamanlı olarak işleyip görselleştiren bir izleme sistemidir. Buradaki asıl teknik zorluk, birden fazla sensörü birbirini bloklamadan okuyabilmektir.

Nasıl Çalışır?

  • 1. Çoklu Sensör Okuma: DHT22 sıcaklık/nem sensörü, LDR ışık sensörü ve MQ-135 hava kalitesi sensöründen sürekli veri alınır. Her sensörün farklı okuma hızı ve gecikme gereksinimi vardır (örn: DHT22 için en az 2 saniye).
  • 2. Veri İşleme — Non-blocking Yaklaşım: Klasik `delay()` kullanımı sistemi durdurur. Bu projede millis() fonksiyonu kullanılarak, sensörlerin birbirini beklemediği, gerçek zamanlı çoklu görev (multitasking) mimarisi kurulmuştur.
  • 3. Görselleştirme: Veriler I2C protokolü ile OLED ekrana aktarılır; sayısal değerlerin yanı sıra bar-grafik ile görsel yoğunluk sağlanır.
  • 4. Eşik Bazlı Uyarı: Değerler normal aralığın dışına çıktığında RGB LED renk değiştirir ve buzzer uyarı verir.

Pin Bağlantıları

  • DHT22: D2 | LDR: A0 | MQ-135: A1
  • RGB LED: D9(K), D10(Y), D11(M) | Buzzer: D6
  • OLED Ekran: I2C (SDA/SCL)

Öğretici Değer (Sunum Notları)

  • I2C Protokolü: Aynı 2 kablolu hat üzerinden birden fazla cihazın adreslenmesi.
  • Non-blocking Programlama: Gerçek zamanlı sistemlerde zamanlayıcıların önemi.
  • Veri Görselleştirme: Ham verilerin insan tarafından okunabilir grafiksel formatlara dönüştürülmesi.

Kaynak Kod

#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <DHT.h>

#define mq_135 A1
#define dht_22 2
#define dht_tipi DHT22
#define ldr A0
#define kirmizi 9
#define yesil 10
#define mavi 11
#define buzzer 6

#define ekran_genislik 128
#define ekran_yukseklik 64
#define ekran_reset -1

DHT dht(dht_22, dht_tipi);
Adafruit_SSD1306 display(ekran_genislik, ekran_yukseklik, &Wire, ekran_reset);

unsigned long son_ldr_okumasi=0;
const unsigned long ldr_okuma_araligi=200;
int anlik_isik=0;

unsigned long son_dht_okumasi=0;
const unsigned long dht_okuma_araligi=2500;
float sicaklik=0.0;
float nem=0.0;

float min_nem=30.0;
float max_nem=80.0;
float min_sicaklik=17.0;
float max_sicaklik=33.0;
bool sensor_hata=false;

unsigned long alarm_ses=0;
const unsigned long alarm_ses_araligi=1000;

void renk_sec(int k,int y,int m){
  analogWrite(kirmizi,k);
  analogWrite(yesil,y);
  analogWrite(mavi,m);
}

void setup() {
  dht.begin();
  Serial.begin(9600);
  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  display.clearDisplay();
  display.setTextColor(SSD1306_WHITE);
  display.setTextSize(1);
  pinMode(kirmizi,OUTPUT);
  pinMode(mavi,OUTPUT);
  pinMode(yesil,OUTPUT);
  pinMode(buzzer,OUTPUT);
}

void loop() {
  unsigned long simdiki_zaman=millis();
  if(simdiki_zaman-son_ldr_okumasi>=ldr_okuma_araligi){
    son_ldr_okumasi=simdiki_zaman;
    anlik_isik=map(analogRead(ldr),0,1023,0,100);
  }
  if(simdiki_zaman-son_dht_okumasi>=dht_okuma_araligi){
    float s=dht.readTemperature();
    float n=dht.readHumidity();
    son_dht_okumasi=simdiki_zaman;
    if(!isnan(s) && !isnan(n)){
      sensor_hata=false;
      sicaklik=s;
      nem=n;
    } else {
      sensor_hata=true;
    }
  }
  if(sensor_hata==true){
    renk_sec(0,0,255);
  } else {
    if((max_nem < nem || nem < min_nem)||(max_sicaklik < sicaklik || sicaklik < min_sicaklik)){
      renk_sec(255,0,0);
      if(simdiki_zaman-alarm_ses>=alarm_ses_araligi){
        alarm_ses=simdiki_zaman;
        tone(buzzer,1500,150);
      }
    } else {
      renk_sec(0,255,0);
    }
  }
}
