Alerji ve Bitki İzolasyon Kabini - ESP32

Giriş ve Proje Amacı

Bitkilerin en sağlıklı şekilde gelişmesini sağlarken, kullanıcı sağlığını alerjen korumasıyla ön planda tutan, ESP32 tabanlı otonom bir izolasyon kabinidir. Sistem; sıcaklık, nem, hava kalitesi (polen) ve ışık şiddetini yüksek doğrulukla kontrol eder.

Literatür Taraması ve Özgün Değer

  • Mevcut Durum: Piyasadaki sistemler çoğunlukla manuel veya temel zamanlayıcılıdır.
  • Özgün Değer: Projemiz, ESP32'nin işlem gücü ile sensör verilerini işleyerek "Hibrit Otomasyon" sağlar.
  • Güvenlik: Erişim kontrollü yetiştirme alanı konsepti ile fiziksel güvenlik ve IoT entegrasyonu sunar.

Proje Planı

  • Aşamalar: Literatür taraması, devre tasarımı, yazılım geliştirme, mekanik tasarım, birleştirme ve test süreçleri toplam 1 yarıyılda tamamlanmıştır.

Blok Şeması

Malzeme Listesi

  • Mikrodenetleyici: ESP32.
  • Sensörler: Kapasitif toprak nem, Sharp GP2Y10 toz, GY302 (BH1750) ışık, DTH22 ısı ve nem.
  • Aktüatörler: Salyangoz fan, dalgıç su pompası, ultrasonik nem yapıcı, tam spektrum LED, Peltier soğutucu.
  • Kontrol: 4 kanal röle modülleri, LM2596 voltaj regülatörü.

Bacak Bağlantı ve Devre Şeması

Teknik Detaylar ve Sensör Mantığı

  • Hava Kalitesi: Sharp GP2Y10 ile partikül tespiti yapılarak HEPA filtre ve fanlar üzerinden sterilizasyon sağlanır.
  • Mikro-İklim: Peltier soğutucu ve ultrasonik nem yapıcı ile ideal tropikal ortam simülasyonu gerçekleştirilir.
  • Kontrol Algoritması: Sistem; sıcaklık >28.5°C, nem <%35 ve polen >0.35mg/m³ gibi eşik değerlerine göre karar verir.

// -------- PIN TANIMLAMALARI --------
#define DHTPIN 27
#define SALYANGOZ_FAN 4
#define TEKLI_FAN 15
#define GROW_RELAY 19
#define POMPA_RELAY 25
#define TOPRAK_NEM_PIN 33
#define BUTTON_PIN 18

Adafruit_SSD1306 display(128, 64, &Wire, -1);
LiquidCrystal_I2C lcd(0x27, 16, 2);
BH1750 lightMeter;
DHT dht(DHTPIN, DHT22);
BlynkTimer timer;

char ssid[] = "Osman";
char pass[] = "gozgozgoztepe";

// -------- DURUMLAR --------
bool turboMode = false;
bool pompaBlynk = false;
bool manualLed = false;

#define AC LOW
#define KAPAT HIGH

// -------- BUTON DEBOUNCE --------
bool sonButonDurumu = HIGH;
unsigned long sonDebounceZamani = 0;
const unsigned long debounceGecikmesi = 50;

// Blynk Kontrolleri
BLYNK_WRITE(V10) { manualLed = param.asInt(); }
BLYNK_WRITE(V11) { pompaBlynk = param.asInt(); }
BLYNK_WRITE(V12) { turboMode = param.asInt(); }

void verileriGonder() {
  float t = dht.readTemperature();
  float h = dht.readHumidity();
  float lux = lightMeter.readLightLevel();
  int nemRaw = analogRead(TOPRAK_NEM_PIN);
  int nemYuzde = constrain(map(nemRaw, 4095, 1500, 0, 100), 0, 100);

  if (Blynk.connected()) {
    Blynk.virtualWrite(V0, t);
    Blynk.virtualWrite(V1, h);
    Blynk.virtualWrite(V2, nemYuzde);
    Blynk.virtualWrite(V3, lux);
  }
}

void butonKontrol() {
  bool okunanDurum = digitalRead(BUTTON_PIN);

  if (okunanDurum != sonButonDurumu) {
    sonDebounceZamani = millis();
  }

  if ((millis() - sonDebounceZamani) > debounceGecikmesi) {
    // Buton basılı (INPUT_PULLUP -> basılı = LOW) ve önceki durumdan farklıysa
    if (okunanDurum == LOW && sonButonDurumu == HIGH) {
      manualLed = !manualLed;
      if (Blynk.connected()) {
        Blynk.virtualWrite(V10, manualLed);
      }
    }
  }
  sonButonDurumu = okunanDurum;
}

void oledGuncelle(float t, float h, int nemYuzde, bool fanAcik, bool pompaAcik, bool ledAcik) {
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);

  display.setCursor(0, 0);
  display.print("Sicaklik: ");
  display.print(t, 1);
  display.println(" C");

  display.setCursor(0, 12);
  display.print("Nem: ");
  display.print(h, 1);
  display.println(" %");

  display.setCursor(0, 24);
  display.print("Toprak: ");
  display.print(nemYuzde);
  display.println(" %");

  display.drawLine(0, 36, 127, 36, SSD1306_WHITE);

  display.setCursor(0, 42);
  display.print("Fan: ");
  display.println(fanAcik ? "ACIK" : "KAPALI");

  display.setCursor(0, 52);
  display.print("Pompa: ");
  display.print(pompaAcik ? "ACIK " : "KAPALI ");
  display.println(ledAcik ? "LED:ON" : "LED:OFF");

  display.display();
}

void setup() {
  Serial.begin(115200);
  Wire.begin();
  dht.begin();
  lightMeter.begin(BH1750::CONTINUOUS_HIGH_RES_MODE);

  lcd.begin();
  lcd.backlight();

  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("OLED baslatilamadi");
  }
  display.clearDisplay();
  display.display();

  pinMode(SALYANGOZ_FAN, OUTPUT);
  pinMode(TEKLI_FAN, OUTPUT);
  pinMode(GROW_RELAY, OUTPUT);
  pinMode(POMPA_RELAY, OUTPUT);
  pinMode(BUTTON_PIN, INPUT_PULLUP);

  // Başlangıç Ayarları (Her şey Kapalı)
  digitalWrite(SALYANGOZ_FAN, KAPAT);
  digitalWrite(TEKLI_FAN, KAPAT);
  digitalWrite(GROW_RELAY, KAPAT);
  digitalWrite(POMPA_RELAY, KAPAT);

  lcd.print("WiFi Baglaniyor");
  Blynk.begin(BLYNK_AUTH_TOKEN, ssid, pass, "blynk.cloud", 80);
  timer.setInterval(3000L, verileriGonder);

  lcd.clear();
  lcd.print("CORESTOFF v20.0");
}

void loop() {
  Blynk.run();
  timer.run();
  butonKontrol();

  float t = dht.readTemperature();
  float h = dht.readHumidity();
  float lux = lightMeter.readLightLevel();
  int nemRaw = analogRead(TOPRAK_NEM_PIN);
  int nemYuzde = constrain(map(nemRaw, 4095, 1500, 0, 100), 0, 100);

  // 1. LED KONTROLÜ
  bool ledAcik = (lux < 50.0 || manualLed);
  digitalWrite(GROW_RELAY, ledAcik ? AC : KAPAT);

  // 2. FAN KONTROLÜ
  bool fanAcik;
  if (turboMode || t > 28.5) {
    digitalWrite(SALYANGOZ_FAN, AC); digitalWrite(TEKLI_FAN, AC);
    fanAcik = true;
  } else if (t < 27.0) {
    digitalWrite(SALYANGOZ_FAN, KAPAT); digitalWrite(TEKLI_FAN, KAPAT);
    fanAcik = false;
  } else {
    fanAcik = (digitalRead(SALYANGOZ_FAN) == AC);
  }

  // 3. POMPA KONTROLÜ (Su Seviyesi Devre Dışı)
  // Kural: Sadece Toprak kuruysa %35 VEYA telefondan (V11) basılmışsa çalış
  bool pompaAcik = (nemYuzde < 35 || pompaBlynk);
  digitalWrite(POMPA_RELAY, pompaAcik ? AC : KAPAT);

  // LCD GÜNCELLEME
  lcd.setCursor(0, 0);
  lcd.print("Lux:"); lcd.print((int)lux);
  lcd.print(Blynk.connected() ? " B:ON" : " B:OFF");

  lcd.setCursor(0, 1);
  lcd.print("Toprak Nem:%"); lcd.print(nemYuzde);

  // OLED GÜNCELLEME
  oledGuncelle(t, h, nemYuzde, fanAcik, pompaAcik, ledAcik);
}
